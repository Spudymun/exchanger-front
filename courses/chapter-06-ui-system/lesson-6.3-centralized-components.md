    setValues(prev => ({ ...prev, [field]: value }));
    if (touched[field] && errors[field]) {
      setErrors(prev => ({ ...prev, [field]: '' }));

      setIsLoading(false);
    }

    setErrors({});
    setTouched({});
  };

    errors,
    touched,
    values,
    handleChange,
    handleSubmit,
